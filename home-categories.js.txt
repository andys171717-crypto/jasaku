const categories = [

{
name:"AC",
icon:"fa-screwdriver-wrench"
},

{
name:"Listrik",
icon:"fa-bolt"
},

{
name:"Laptop",
icon:"fa-laptop"
},

{
name:"CCTV",
icon:"fa-video"
},

{
name:"Lainnya",
icon:"fa-ellipsis"
}

];

const kategoriList =
document.getElementById(
"kategoriList"
);

if(kategoriList){

kategoriList.innerHTML =
categories.map(item => `

<div
class="card category-card"
data-category="${item.name}">

<i class="fas ${item.icon}"></i>

<span>
${item.name}
</span>

</div>

`).join("");



document
.querySelectorAll(
".category-card"
)
.forEach(card=>{

card.addEventListener(
"click",
()=>{

const kategori =
card.dataset.category;

document
.querySelectorAll(
".category-card"
)
.forEach(c=>{

c.classList.remove(
"active-category"
);

});

card.classList.add(
"active-category"
);

const searchInput =
document.getElementById(
"searchInput"
);

if(!searchInput) return;

if(
kategori ===
"Lainnya"
){

searchInput.value =
"Lainnya";

}else{

searchInput.value =
kategori;

}

document
.getElementById(
"searchBtn"
)
.click();

document
.getElementById(
"servicesList"
)
.scrollIntoView({

behavior:"smooth"

});

});

});

}
